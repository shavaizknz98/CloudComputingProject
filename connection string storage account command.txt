-- Create USER table
CREATE TABLE USERTABLE
(
    EMAIL NVARCHAR(255) PRIMARY KEY,
    NAME NVARCHAR(70) NOT NULL,
    PASS NVARCHAR(50) NOT NULL,
);

-- Create VIDEOINFO table
CREATE TABLE VIDEOINFO
(
    VIDID INT IDENTITY NOT NULL PRIMARY KEY,
    VIDPATH NVARCHAR(500) NOT NULL,
    EMAIL NVARCHAR(255) REFERENCES USERTABLE (EMAIL),
    VIDPASS NVARCHAR(70) NOT NULL,
    VIDNAME NVARCHAR(70) NOT NULL,
    ANALYZED BIT NOT NULL,
    CDATE NVARCHAR(255) NOT NULL

);
